## 2) 증감연산자
++ : a = a + 1 (1씩 증가)
-- : a = a - 1 (1씩 감소)

a++; // 다른 연산자 먼저 모두 계산하고 - 마지막에 ++ 
++a; // 가장 먼저 ++하고 나머지 연산을 진행

ex) resutl = a++ * 20
'''
a = 10
result = 10 * 20
a = a + 1  // 11

result = 200
a = 11
'''

e2) resutl = ++a * 20
'''
a = 10
a = 10 + 1
result = 11 * 20

result = 220
a = 11
'''
